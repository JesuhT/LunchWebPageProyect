-- Creación de las tablas
CREATE TABLE Roles (
    ID_rol INT AUTO_INCREMENT PRIMARY KEY,
    Nombre VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE Almuerzos (
    ID_almuerzo INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    descripcion VARCHAR(255) NOT NULL
);

CREATE TABLE Calificaciones (
    ID_calificación INT AUTO_INCREMENT PRIMARY KEY,
    ID_estudiante INT NOT NULL,
    ID_almuerzo INT NOT NULL,
    calificación INT NOT NULL,
    descripción VARCHAR(255),
    fecha_calificación DATE NOT NULL
);

CREATE TABLE Filas_Virtuales (
    ID_fila INT AUTO_INCREMENT PRIMARY KEY,
    fecha_realización DATE NOT NULL,
    num_personas INT NOT NULL
);

CREATE TABLE Usuarios (
    ID_user INT AUTO_INCREMENT PRIMARY KEY,
    Nombre VARCHAR(32) NOT NULL,
    Apellido VARCHAR(32) NOT NULL,
    Email VARCHAR(32) UNIQUE NOT NULL,
    Contrasena VARCHAR(32) NOT NULL,
    Celular VARCHAR(20) UNIQUE NOT NULL,
    ID_programa INT,
    ID_rol INT DEFAULT 1
);

CREATE TABLE Días_almuerzos (
    ID_Día INT PRIMARY KEY,
    nombre VARCHAR(32) UNIQUE NOT NULL
);

CREATE TABLE Facultades (
    ID_facultad INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(200) UNIQUE NOT NULL
);

CREATE TABLE Días_Almuerzos_Estudiantes (
    ID_dia INT NOT NULL,
    ID_estudiante INT NOT NULL
);

CREATE TABLE Menus (
    ID_menu INT AUTO_INCREMENT PRIMARY KEY,
    ID_dia INT NOT NULL
);

CREATE TABLE Programas (
    ID_programa INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(32) UNIQUE NOT NULL,
    ID_facultad INT NOT NULL
);

CREATE TABLE Donaciones (
    ID_donación INT AUTO_INCREMENT PRIMARY KEY,
    ID_donante INT NOT NULL,
    ID_beneficiario INT NOT NULL,
    fecha DATE NOT NULL,
    hora TIME NOT NULL,
    ID_estado INT NOT NULL
);

CREATE TABLE Estados (
    ID_estado INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(32) NOT NULL
);

CREATE TABLE Estados_Donaciones (
    ID_donación INT NOT NULL,
    ID_estado INT NOT NULL
);

CREATE TABLE Modificaciones_Menus (
    ID_administrador INT NOT NULL,
    ID_menu INT NOT NULL,
    fecha DATE NOT NULL,
    hora TIME NOT NULL
);

CREATE TABLE Almuerzos_En_Menus (
    ID_menu INT NOT NULL,
    ID_almuerzo INT NOT NULL
);

CREATE TABLE Estudiantes_En_Fila (
    ID_estudiante INT NOT NULL,
    ID_fila INT NOT NULL,
    turno INT NOT NULL,
    hora_ingreso TIME NOT NULL
);

-- agregar restricciones de clave foránea
ALTER TABLE Calificaciones
ADD CONSTRAINT Calificacion_fk0 FOREIGN KEY (ID_estudiante) REFERENCES Usuarios(ID_user) ON DELETE CASCADE,
ADD CONSTRAINT Calificacion_fk1 FOREIGN KEY (ID_almuerzo) REFERENCES Almuerzos(ID_almuerzo) ON DELETE CASCADE;

ALTER TABLE Menus
ADD CONSTRAINT Menu_fk0 FOREIGN KEY (ID_menu) REFERENCES Días_almuerzos(ID_Día) ON DELETE CASCADE;

ALTER TABLE Días_Almuerzos_Estudiantes
ADD CONSTRAINT Dias_Almuerzos_Estudiantes_fk0 FOREIGN KEY (ID_dia) REFERENCES Días_almuerzos(ID_Día) ON DELETE CASCADE,
ADD CONSTRAINT Dias_Almuerzos_Estudiantes_fk1 FOREIGN KEY (ID_estudiante) REFERENCES Usuarios(ID_user) ON DELETE CASCADE;

ALTER TABLE Programas
ADD CONSTRAINT Programa_fk0 FOREIGN KEY (ID_facultad) REFERENCES Facultades(ID_facultad) ON DELETE CASCADE;

ALTER TABLE Donaciones
ADD CONSTRAINT Donación_fk0 FOREIGN KEY (ID_donante) REFERENCES Usuarios(ID_user) ON DELETE CASCADE,
ADD CONSTRAINT Donación_fk1 FOREIGN KEY (ID_beneficiario) REFERENCES Usuarios(ID_user) ON DELETE CASCADE,
ADD CONSTRAINT Donación_fk2 FOREIGN KEY (ID_estado) REFERENCES Estados(ID_estado) ON DELETE CASCADE;

ALTER TABLE Estados_Donaciones
ADD CONSTRAINT Estados_Donaciones_fk0 FOREIGN KEY (ID_donación) REFERENCES Donaciones(ID_donación) ON DELETE CASCADE,
ADD CONSTRAINT Estados_Donaciones_fk1 FOREIGN KEY (ID_estado) REFERENCES Estados(ID_estado) ON DELETE CASCADE;

ALTER TABLE Modificaciones_Menus
ADD CONSTRAINT Modificaciones_Menu_fk0 FOREIGN KEY (ID_administrador) REFERENCES Usuarios(ID_user) ON DELETE CASCADE,
ADD CONSTRAINT Modificaciones_Menu_fk1 FOREIGN KEY (ID_menu) REFERENCES Menus(ID_menu) ON DELETE CASCADE;

ALTER TABLE Almuerzos_En_Menus
ADD CONSTRAINT Almuerzos_En_Menu_fk0 FOREIGN KEY (ID_menu) REFERENCES Menus(ID_menu) ON DELETE CASCADE,
ADD CONSTRAINT Almuerzos_En_Menu_fk1 FOREIGN KEY (ID_almuerzo) REFERENCES Almuerzos(ID_almuerzo) ON DELETE CASCADE;

ALTER TABLE Estudiantes_En_Fila
ADD CONSTRAINT Estudiantes_En_Fila_fk0 FOREIGN KEY (ID_estudiante) REFERENCES Usuarios(ID_user) ON DELETE CASCADE,
ADD CONSTRAINT Estudiantes_En_Fila_fk1 FOREIGN KEY (ID_fila) REFERENCES Filas_Virtuales(ID_fila);

ALTER TABLE Usuarios
ADD CONSTRAINT Estudiantes_fk0 FOREIGN KEY (ID_programa) REFERENCES Programas(ID_programa) ON DELETE CASCADE,
ADD CONSTRAINT Estudiantes_fk1 FOREIGN KEY (ID_rol) REFERENCES Roles(ID_rol) ON DELETE CASCADE;
